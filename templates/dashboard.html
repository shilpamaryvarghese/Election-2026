async function loadTable(){ let tbody = document.getElementById("tableBody"); tbody.innerHTML = `Loading...`; try { let res = await fetch("/api/constituency/all"); let json = await res.json(); tbody.innerHTML = ""; if(!json.data || json.data.length === 0){ tbody.innerHTML = `No data found`; return; } json.data.forEach(r => { tbody.innerHTML += ` ${r.constituency} ${r.const_no} ${r.leading_candidate} ${r.leading_party} ${r.trailing_candidate} ${r.trailing_party} ${r.margin} ${r.round} ${r.status} `; }); } catch (err){ tbody.innerHTML = `Error loading data`; console.error(err); } }
